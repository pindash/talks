    def max_profit(prices: list[int], k=1) -> int:
        @lru_cache(None)
        def rec(i, k, f):
            if k == 0 or i == len(prices):
                return 0
            c1 = rec(i+1, k, f) # we dont buy or sell
            c2 = 0
            if f:
                c2 = rec(i+1, k-1, False) + prices[i]  # we sell
            else:
                c2 = rec(i+1, k, True) - prices[i]  # we buy
            return max(c1, c2)
        return rec(0, k, False)  # k == 1, means, can do only 1 transaction

    #prefixes top down version
    def max_profit(prices: list[int], k=1) -> int:
        @lru_cache(None)
        def rec(i, k, h):
                if k == 0:
                    return 0
                if i == 0:
                    return h*-prices[0]
                c1 = rec(i-1, k, h) # we dont buy or sell
                c2 = 0
                if h:
                    c2 = rec(i-1, k-1, False) - prices[i]  # we buy
                else:
                    c2 = rec(i-1, k, True) + prices[i]  # we sell
                return max(c1, c2)
        return rec(len(prices)-1, k, False)  # k == 1, means, can do only 1 transaction
